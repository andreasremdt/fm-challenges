import { Meta, ColorPalette, ColorItem } from "@storybook/blocks";

import { theme } from "../../tailwind.config.js";

<Meta title="Colors" />

# Colors

<ColorPalette>
  {Object.keys(theme.colors).map((color) => {
    if (typeof theme.colors[color] === "string") {
      return <ColorItem title={color} colors={[theme.colors[color]]} key={color} />
    }

    return (
      <ColorItem title={color} colors={theme.colors[color]} />
    )

})}

</ColorPalette>
